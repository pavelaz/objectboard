package com.psg.objectboard.model.own.ownsEntity.classVO.OtherVO;

import java.util.ArrayList;

public class AssignmentsReportVO {
    private AdditionalDataReportVO additionalDataReport;
    private ArrayList<AssignmentsConsultVO> assignmentsConsult;

    public AssignmentsReportVO() {
    }

    public AssignmentsReportVO(AdditionalDataReportVO additionalDataReport, ArrayList<AssignmentsConsultVO> assignmentsConsult) {
        this.additionalDataReport = additionalDataReport;
        this.assignmentsConsult = assignmentsConsult;
    }

    public AdditionalDataReportVO getAdditionalDataReport() {
        return additionalDataReport;
    }

    public void setAdditionalDataReport(AdditionalDataReportVO additionalDataReport) {
        this.additionalDataReport = additionalDataReport;
    }

    public ArrayList<AssignmentsConsultVO> getAssignmentsConsult() {
        return assignmentsConsult;
    }

    public void setAssignmentsConsult(ArrayList<AssignmentsConsultVO> assignmentsConsult) {
        this.assignmentsConsult = assignmentsConsult;
    }
}
