package com.psg.objectboard.model.entity;

import javax.persistence.*;
import java.util.Collection;
import java.util.Objects;

@Entity
@Table(name = "bussinessUnit", schema = "objectboard_db", catalog = "")
public class BussinessUnitEntity {
    private long buBisCode;
    private String buFederalNumber;
    private String buProvincialNumber;
    private String buEmail;
    private String buStatus;
    private String buSuperCode;
    private long countryCoCountryCode;
    private long statesStStateCode;
    private long statesCountryCoCountryCode;
    private long cityCiCityCode;
    private long cityStatesStStateCode;
    private long cityStatesCountryCoCountryCode;
    private long bussinessTypeBtCodeType;
    private CountryEntity countryByCountryCoCountryCode;
    private StatesEntity states;
    private CityEntity city;
    private BussinessTypeEntity bussinessTypeByBussinessTypeBtCodeType;
    private Collection<BussinessUnitHasCommonDataEntity> bussinessUnitHasCommonDataByBuBisCode;
    private Collection<DischargeEntity> dischargesByBuBisCode;
    private Collection<MasterUserEntity> masterUsersByBuBisCode;

    @Id
    @Column(name = "bu_bis_code")
    public long getBuBisCode() {
        return buBisCode;
    }

    public void setBuBisCode(long buBisCode) {
        this.buBisCode = buBisCode;
    }

    @Basic
    @Column(name = "bu_federal_number")
    public String getBuFederalNumber() {
        return buFederalNumber;
    }

    public void setBuFederalNumber(String buFederalNumber) {
        this.buFederalNumber = buFederalNumber;
    }

    @Basic
    @Column(name = "bu_provincial_number")
    public String getBuProvincialNumber() {
        return buProvincialNumber;
    }

    public void setBuProvincialNumber(String buProvincialNumber) {
        this.buProvincialNumber = buProvincialNumber;
    }

    @Basic
    @Column(name = "bu_email")
    public String getBuEmail() {
        return buEmail;
    }

    public void setBuEmail(String buEmail) {
        this.buEmail = buEmail;
    }

    @Basic
    @Column(name = "bu_status")
    public String getBuStatus() {
        return buStatus;
    }

    public void setBuStatus(String buStatus) {
        this.buStatus = buStatus;
    }

    @Basic
    @Column(name = "bu_super_code")
    public String getBuSuperCode() {
        return buSuperCode;
    }

    public void setBuSuperCode(String buSuperCode) {
        this.buSuperCode = buSuperCode;
    }

    @Basic
    @Column(name = "country_co_country_code")
    public long getCountryCoCountryCode() {
        return countryCoCountryCode;
    }

    public void setCountryCoCountryCode(long countryCoCountryCode) {
        this.countryCoCountryCode = countryCoCountryCode;
    }

    @Basic
    @Column(name = "states_st_state_code")
    public long getStatesStStateCode() {
        return statesStStateCode;
    }

    public void setStatesStStateCode(long statesStStateCode) {
        this.statesStStateCode = statesStStateCode;
    }

    @Basic
    @Column(name = "states_country_co_country_code")
    public long getStatesCountryCoCountryCode() {
        return statesCountryCoCountryCode;
    }

    public void setStatesCountryCoCountryCode(long statesCountryCoCountryCode) {
        this.statesCountryCoCountryCode = statesCountryCoCountryCode;
    }

    @Basic
    @Column(name = "city_ci_city_code")
    public long getCityCiCityCode() {
        return cityCiCityCode;
    }

    public void setCityCiCityCode(long cityCiCityCode) {
        this.cityCiCityCode = cityCiCityCode;
    }

    @Basic
    @Column(name = "city_states_st_state_code")
    public long getCityStatesStStateCode() {
        return cityStatesStStateCode;
    }

    public void setCityStatesStStateCode(long cityStatesStStateCode) {
        this.cityStatesStStateCode = cityStatesStStateCode;
    }

    @Basic
    @Column(name = "city_states_country_co_country_code")
    public long getCityStatesCountryCoCountryCode() {
        return cityStatesCountryCoCountryCode;
    }

    public void setCityStatesCountryCoCountryCode(long cityStatesCountryCoCountryCode) {
        this.cityStatesCountryCoCountryCode = cityStatesCountryCoCountryCode;
    }

    @Basic
    @Column(name = "bussinessType_bt_code_type")
    public long getBussinessTypeBtCodeType() {
        return bussinessTypeBtCodeType;
    }

    public void setBussinessTypeBtCodeType(long bussinessTypeBtCodeType) {
        this.bussinessTypeBtCodeType = bussinessTypeBtCodeType;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        BussinessUnitEntity that = (BussinessUnitEntity) o;
        return buBisCode == that.buBisCode &&
                countryCoCountryCode == that.countryCoCountryCode &&
                statesStStateCode == that.statesStStateCode &&
                statesCountryCoCountryCode == that.statesCountryCoCountryCode &&
                cityCiCityCode == that.cityCiCityCode &&
                cityStatesStStateCode == that.cityStatesStStateCode &&
                cityStatesCountryCoCountryCode == that.cityStatesCountryCoCountryCode &&
                bussinessTypeBtCodeType == that.bussinessTypeBtCodeType &&
                Objects.equals(buFederalNumber, that.buFederalNumber) &&
                Objects.equals(buProvincialNumber, that.buProvincialNumber) &&
                Objects.equals(buEmail, that.buEmail) &&
                Objects.equals(buStatus, that.buStatus) &&
                Objects.equals(buSuperCode, that.buSuperCode);
    }

    @Override
    public int hashCode() {
        return Objects.hash(buBisCode, buFederalNumber, buProvincialNumber, buEmail, buStatus, buSuperCode, countryCoCountryCode, statesStStateCode, statesCountryCoCountryCode, cityCiCityCode, cityStatesStStateCode, cityStatesCountryCoCountryCode, bussinessTypeBtCodeType);
    }

    @ManyToOne
    @JoinColumn(name = "country_co_country_code", referencedColumnName = "co_country_code", nullable = false)
    public CountryEntity getCountryByCountryCoCountryCode() {
        return countryByCountryCoCountryCode;
    }

    public void setCountryByCountryCoCountryCode(CountryEntity countryByCountryCoCountryCode) {
        this.countryByCountryCoCountryCode = countryByCountryCoCountryCode;
    }

    @ManyToOne
    @JoinColumns({@JoinColumn(name = "states_st_state_code", referencedColumnName = "st_state_code", nullable = false), @JoinColumn(name = "states_country_co_country_code", referencedColumnName = "country_co_country_code", nullable = false)})
    public StatesEntity getStates() {
        return states;
    }

    public void setStates(StatesEntity states) {
        this.states = states;
    }

    @ManyToOne
    @JoinColumns({@JoinColumn(name = "city_ci_city_code", referencedColumnName = "ci_city_code", nullable = false), @JoinColumn(name = "city_states_st_state_code", referencedColumnName = "states_st_state_code", nullable = false), @JoinColumn(name = "city_states_country_co_country_code", referencedColumnName = "states_country_co_country_code", nullable = false)})
    public CityEntity getCity() {
        return city;
    }

    public void setCity(CityEntity city) {
        this.city = city;
    }

    @ManyToOne
    @JoinColumn(name = "bussinessType_bt_code_type", referencedColumnName = "bt_code_type", nullable = false)
    public BussinessTypeEntity getBussinessTypeByBussinessTypeBtCodeType() {
        return bussinessTypeByBussinessTypeBtCodeType;
    }

    public void setBussinessTypeByBussinessTypeBtCodeType(BussinessTypeEntity bussinessTypeByBussinessTypeBtCodeType) {
        this.bussinessTypeByBussinessTypeBtCodeType = bussinessTypeByBussinessTypeBtCodeType;
    }

    @OneToMany(mappedBy = "bussinessUnitByBussinessUnitBuBisCode")
    public Collection<BussinessUnitHasCommonDataEntity> getBussinessUnitHasCommonDataByBuBisCode() {
        return bussinessUnitHasCommonDataByBuBisCode;
    }

    public void setBussinessUnitHasCommonDataByBuBisCode(Collection<BussinessUnitHasCommonDataEntity> bussinessUnitHasCommonDataByBuBisCode) {
        this.bussinessUnitHasCommonDataByBuBisCode = bussinessUnitHasCommonDataByBuBisCode;
    }

    @OneToMany(mappedBy = "bussinessUnitByBussinessUnitBuBisCode")
    public Collection<DischargeEntity> getDischargesByBuBisCode() {
        return dischargesByBuBisCode;
    }

    public void setDischargesByBuBisCode(Collection<DischargeEntity> dischargesByBuBisCode) {
        this.dischargesByBuBisCode = dischargesByBuBisCode;
    }

    @OneToMany(mappedBy = "bussinessUnitByBussinessUnitBuBisCode")
    public Collection<MasterUserEntity> getMasterUsersByBuBisCode() {
        return masterUsersByBuBisCode;
    }

    public void setMasterUsersByBuBisCode(Collection<MasterUserEntity> masterUsersByBuBisCode) {
        this.masterUsersByBuBisCode = masterUsersByBuBisCode;
    }
}
