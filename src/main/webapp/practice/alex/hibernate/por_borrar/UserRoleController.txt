package com.psg.objectboard.controller;

import com.psg.objectboard.model.service.UserRoleService;

import java.util.Scanner;

public class UserRoleController {
    private UserRoleService userRoleService;

    public UserRoleController(){this.userRoleService = new UserRoleService();}

    public void sampleDetailUserRole(){
        Scanner scanner=new Scanner(System.in);
        System.out.println("Quel est l'identifiant de la langue dont vous voulez afficher les informations ?");
        long identifier=scanner.nextLong();
        UserRoleDto userRoleDto= userRoleService.getUserRole(identifier);
        System.out.println("La langue sélectionné s'applle "+ userRoleDto.getMasterUserMuEmail());
    }
}
