package com.psg.objectboard.controller;

import com.psg.objectboard.model.service.StatesService;

import java.util.Scanner;

public class StatesController {
    private StatesService statesService;

    public StatesController(){this.statesService = new StatesService();}

    public void sampleDetailStates(){
        Scanner scanner=new Scanner(System.in);
        System.out.println("Quel est l'identifiant de la langue dont vous voulez afficher les informations ?");
        long identifier=scanner.nextLong();
        StatesDto statesDto= statesService.getStates(identifier);
        System.out.println("La langue sélectionné s'applle "+ statesDto.getStName());
    }
}
