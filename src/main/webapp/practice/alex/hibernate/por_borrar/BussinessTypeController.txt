import java.util.Scanner;

public class BussinessTypeController {
    private BussinessTypeService bussinessTypeService;

    public BussinessTypeController(){this.bussinessTypeService = new BussinessTypeService();}

    public void sampleDetailBussinessType(){

        System.out.println("Quel est l'identifiant de la langue dont vous voulez afficher les informations ?");
        long identifier=scanner.nextLong();

        System.out.println("La langue sélectionné s'applle "+ bussinessTypeDto.getBtDescription());
    }
}
