package com.psg.objectboard.model.entity;

import javax.persistence.*;
import java.util.Collection;
import java.util.Objects;

@Entity
@Table(name = "country", schema = "objectboard_db", catalog = "")
public class CountryEntity {
    private long coCountryCode;
    private String coName;
    private Collection<BussinessUnitEntity> bussinessUnitsByCoCountryCode;
    private Collection<StatesEntity> statesByCoCountryCode;

    @Id
    @Column(name = "co_country_code")
    public long getCoCountryCode() {
        return coCountryCode;
    }

    public void setCoCountryCode(long coCountryCode) {
        this.coCountryCode = coCountryCode;
    }

    @Basic
    @Column(name = "co_name")
    public String getCoName() {
        return coName;
    }

    public void setCoName(String coName) {
        this.coName = coName;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        CountryEntity that = (CountryEntity) o;
        return coCountryCode == that.coCountryCode &&
                Objects.equals(coName, that.coName);
    }

    @Override
    public int hashCode() {
        return Objects.hash(coCountryCode, coName);
    }

    @OneToMany(mappedBy = "countryByCountryCoCountryCode")
    public Collection<BussinessUnitEntity> getBussinessUnitsByCoCountryCode() {
        return bussinessUnitsByCoCountryCode;
    }

    public void setBussinessUnitsByCoCountryCode(Collection<BussinessUnitEntity> bussinessUnitsByCoCountryCode) {
        this.bussinessUnitsByCoCountryCode = bussinessUnitsByCoCountryCode;
    }

    @OneToMany(mappedBy = "countryByCountryCoCountryCode")
    public Collection<StatesEntity> getStatesByCoCountryCode() {
        return statesByCoCountryCode;
    }

    public void setStatesByCoCountryCode(Collection<StatesEntity> statesByCoCountryCode) {
        this.statesByCoCountryCode = statesByCoCountryCode;
    }
}
