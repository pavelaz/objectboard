package com.psg.objectboard.model.service;
import com.psg.objectboard.model.EntityManagerHolder;
import com.psg.objectboard.model.repository.CommonDataRepositoryImpl;

import javax.persistence.EntityManager;
import javax.persistence.EntityTransaction;

public class CommonDataService {
    private CommonDataRepositoryImpl commonDataRepository;

    public CommonDataService() {
        this.commonDataRepository = new CommonDataRepositoryImpl();
    }

    public CommonDataService(CommonDataRepositoryImpl countryRepository) {
        this.commonDataRepository = commonDataRepository;
    }

    public CommonDataDto getCommonData(Long id){
        EntityManager entityManager=null;
        EntityTransaction transaction = null;
        CommonDataEntity commonDataEntity= null;
        CommonDataDto commonDataDto = null;
        try{
            entityManager = new EntityManagerHolder().getCurrentEntityManager();
            transaction=entityManager.getTransaction();
            transaction.begin();
            commonDataEntity = commonDataRepository.getCdCode(id);
            commonDataDto = new CommonDataDto();
            commonDataDto.setCdCode(commonDataEntity.getCdCode());
            commonDataDto.setCdName(commonDataEntity.getCdName());
            commonDataDto.setCdAddress(commonDataEntity.getCdAddress());
            commonDataDto.setCdPhone(commonDataEntity.getCdPhone());
            commonDataDto.setCdCelPhone(commonDataEntity.getCdCelPhone());
            commonDataDto.setCdWebpage(commonDataEntity.getCdWebpage());
            commonDataDto.setCdZipcode(commonDataEntity.getCdZipcode());
            transaction.commit();
        }
        catch (Exception e){
            if (transaction!=null)
                transaction.rollback();
            e.printStackTrace();
            System.err.println("Country no funciona");
        }
        finally {
            if (entityManager!=null)
                entityManager.close();
        }
        return commonDataDto;
    }
}
