package com.psg.objectboard.model.service;
import com.psg.objectboard.model.EntityManagerHolder;
import com.psg.objectboard.model.repository.StatesRepositoryImpl;

import javax.persistence.EntityManager;
import javax.persistence.EntityTransaction;

public class StatesService {
    private StatesRepositoryImpl statesRepository;

    public StatesService() {
        this.statesRepository = new StatesRepositoryImpl();
    }

    public StatesService(StatesRepositoryImpl statesRepository) {
        this.statesRepository = statesRepository;
    }

    public StatesDto getStates(Long id){
        EntityManager entityManager=null;
        EntityTransaction transaction = null;
        StatesEntity statesEntity= null;
        StatesDto statesDto = null;
        try{
            entityManager = new EntityManagerHolder().getCurrentEntityManager();
            transaction=entityManager.getTransaction();
            transaction.begin();
            statesEntity = statesRepository.getById(id);
            statesDto = new StatesDto();
            statesDto.setStStateCode(statesEntity.getStStateCode());
            statesDto.setCountryCoCountryCode(statesEntity.getCountryCoCountryCode());
            statesDto.setStName(statesEntity.getStName());
            transaction.commit();
        }
        catch (Exception e){
            if (transaction!=null)
                transaction.rollback();
            e.printStackTrace();
            System.err.println("States no funciona");
        }
        finally {
            if (entityManager!=null)
                entityManager.close();
        }
        return statesDto;
    }
}
