package com.psg.objectboard.controller.resourcesApiRESTful;

import javax.servlet.ServletException;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.ws.rs.*;
import javax.ws.rs.core.Context;
import javax.ws.rs.core.MediaType;
import java.io.IOException;

@Path("/commentsresources")
public class CommentsResource {

    @GET//regresa una lista de libros
    @Path("/insertAdjacentHTML")
    @Produces(MediaType.TEXT_HTML)
    public void insertAdjacentHTML(@Context HttpServletRequest request, @Context HttpServletResponse response) throws ServletException, IOException {
        request.getRequestDispatcher("/WEB-INF/pages/jsp/insertAdjacentHTML/insertNewTr.jsp").forward(request, response);
    }
}
